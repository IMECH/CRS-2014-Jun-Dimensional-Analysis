\begin{problem}[07]
什么是动态(动力学), 静态(静力学), 准静态.
\end{problem}
% --------------------------------------------------------------------
\begin{solution}
\begin{itemize}
\item \textbf{动\hphantom{动}态}: 非平衡的问题是动力学或动态问题.
\item \textbf{静\hphantom{静}态}: 平衡的问题是静力学或静态问题.
\item \textbf{准静态}: 准静态是指动态过程可看成无限多个变化非常慢的过程的组合, 这要求系统发生的变化无限慢.
\end{itemize}
\end{solution}
